CREATE OR REPLACE TABLE Professores (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    localizacao VARCHAR(100) NOT NULL,

    tem_computador BOOLEAN NOT NULL,

    modulo_1 BOOLEAN NOT NULL,
    modulo_kids BOOLEAN NOT NULL,
    modulo_conv BOOLEAN NOT NULL,
    modulo_vip BOOLEAN NOT NULL
